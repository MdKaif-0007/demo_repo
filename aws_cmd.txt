#!/bin/bash 
apt-get update 
apt-get upgrade 
apt-get install -y nginx 
systemctl start nginx 
systemctl enable nginx

curl -sL https://deb.nodesource.com/setup_18.x | sudo -E bash - 
apt-get install -y nodejs
git clone https://github.com/MdKaif-0007/kaif54_repo.git
cd kaif54_repo
npm install
node index.js
